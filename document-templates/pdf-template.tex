%% ── System-font cascade (must come after common.latex's \usepackage{lmodern}) ──
%% Each role tries a designed-for-screen font first, falling back through fonts
%% guaranteed preinstalled on macOS or Windows. If nothing matches, fontspec
%% keeps Latin Modern — no error, just a less polished look.
$if(mainfont)$
$else$
%% Body font — modern humanist sans. Mapping=tex-text lets fontspec substitute
%% glyphs the font lacks (e.g. → arrows in Charter) from the math font.
%%   Avenir Next (macOS)  →  Calibri (Windows)  →  Helvetica Neue (macOS)
%%   →  Helvetica (macOS)  →  Arial (both)
\IfFontExistsTF{Avenir Next}{%
  \setmainfont{Avenir Next}[Ligatures=TeX,Mapping=tex-text]%
}{\IfFontExistsTF{Calibri}{%
  \setmainfont{Calibri}[Ligatures=TeX,Mapping=tex-text]%
}{\IfFontExistsTF{Helvetica Neue}{%
  \setmainfont{Helvetica Neue}[Ligatures=TeX,Mapping=tex-text]%
}{\IfFontExistsTF{Helvetica}{%
  \setmainfont{Helvetica}[Ligatures=TeX,Mapping=tex-text]%
}{\IfFontExistsTF{Arial}{%
  \setmainfont{Arial}[Ligatures=TeX,Mapping=tex-text]%
}{}}}}}

%% Sans family — set the same as the main font so \sffamily and headings stay
%% consistent. (We deliberately do NOT set \setromanfont, because in fontspec
%% it's an alias for \setmainfont and would overwrite our body font.)
\IfFontExistsTF{Avenir Next}{%
  \setsansfont{Avenir Next}[Ligatures=TeX,Mapping=tex-text]%
}{\IfFontExistsTF{Calibri}{%
  \setsansfont{Calibri}[Ligatures=TeX,Mapping=tex-text]%
}{\IfFontExistsTF{Helvetica Neue}{%
  \setsansfont{Helvetica Neue}[Ligatures=TeX,Mapping=tex-text]%
}{\IfFontExistsTF{Helvetica}{%
  \setsansfont{Helvetica}[Ligatures=TeX,Mapping=tex-text]%
}{\IfFontExistsTF{Arial}{%
  \setsansfont{Arial}[Ligatures=TeX,Mapping=tex-text]%
}{}}}}}

%% Glyph fallback: humanist sans body fonts often lack arrows. Route them to
%% a font that has them so they render instead of as the .notdef glyph (?).
%% Apple Symbols ships with macOS; on Windows, Segoe UI Symbol covers them.

  \newfontfamily{\symbolFallback}{Apple Symbols}%
  \def\useSymbolFallback{1}%
}{\IfFontExistsTF{Segoe UI Symbol}{%
  \newfontfamily{\symbolFallback}{Segoe UI Symbol}%
  \def\useSymbolFallback{1}%
}{}}
\ifdefined\useSymbolFallback
  \catcode`\→=\active \def→{{\symbolFallback\char"2192}}%
  \catcode`\←=\active \def←{{\symbolFallback\char"2190}}%
  \catcode`\↑=\active \def↑{{\symbolFallback\char"2191}}%
  \catcode`\↓=\active \def↓{{\symbolFallback\char"2193}}%

%% ── Code-block styling (no external dependencies) ──
%% Pandoc wraps highlighted code as \begin{Shaded}\begin{Highlighting}…
%% \end{Highlighting}\end{Shaded}, where Highlighting is a fancyvrb Verbatim
%% (fancyvrb ships with BasicTeX). We redefine \FancyVerbFormatLine so each line
%% is painted with a \colorbox — this gives a true page-breakable background
%% without needing framed/mdframed/tcolorbox. The left accent stripe is layered
%% on via fancyvrb's frame=leftline.
\DefineVerbatimEnvironment{Highlighting}{Verbatim}{%
  commandchars=\\\{\},%
  frame=leftline,%
  framerule=2.5pt,%
  rulecolor=\color{codeborder},%
  framesep=8pt,%
  xleftmargin=2pt,%
  fontsize=\small,%
  formatcom=\HighlightingFormatLine%
}
%% Per-line background: each line is wrapped in a \colorbox so the fill flows
%% across page breaks. Setting \fboxsep=0pt eliminates the vertical gap between
%% adjacent boxes; a stretched \strut gives the line its full height so the
%% boxes touch and form a continuous fill.
\newcommand{\HighlightingFormatLine}{%
  \setlength{\fboxsep}{0pt}%
  \def\FancyVerbFormatLine##1{%
    \colorbox{shadecolor}{%
      \makebox[\dimexpr\linewidth-12pt][l]{\rule[-3pt]{0pt}{14pt}\hspace{6pt}##1\hspace{6pt}}%
    }%
  }%
}
%% pandoc only defines the Shaded environment via common.latex when the
%% document actually contains highlighted code. For a document without code
%% blocks it never exists, so \renewenvironment would fail with "Environment
%% Shaded undefined". Define-or-redefine: create it if absent, otherwise renew.
\makeatletter
\@ifundefined{Shaded}{%
  \newenvironment{Shaded}{\par\vspace{4pt}}{\par\vspace{4pt}}%
}{%
  \renewenvironment{Shaded}{\par\vspace{4pt}}{\par\vspace{4pt}}%
}
\makeatother

%% ── Better horizontal divider ──
%% Pandoc emits HRs as \begin{center}\rule{0.5\linewidth}{0.5pt}\end{center}.
%% Redefining \rule globally would affect other uses, so we patch it only when it
%% looks like pandoc's HR signature by wrapping in a smarter macro.
\let\pandocOriginalRule\rule
\renewcommand{\rule}[3][0pt]{%
  \ifdim#2=0.5\linewidth\relax
    {\centering
     \color{rulecol}\pandocOriginalRule[#1]{0.18\linewidth}{0.6pt}%
     \hspace{0.6em}\raisebox{-0.18ex}{\textcolor{accent}{\ensuremath{\bullet}}}\hspace{0.6em}%
     \color{rulecol}\pandocOriginalRule[#1]{0.18\linewidth}{0.6pt}\par}%
  \else
    \pandocOriginalRule[#1]{#2}{#3}%
  \fi
}

%% ── Table styling ──
%% Editorial style: full-width tables, generous padding, coloured booktabs
%% rules. No row fills (which avoids the fill-vs-rule width misalignment), no
%% header text styling (which would require fragile per-row patching).
%% Everything used is in BasicTeX (longtable, booktabs, array, etoolbox).
\usepackage{etoolbox}
\usepackage{colortbl}                       % for \arrayrulecolor
\definecolor{tableRule}{HTML}{4F46E5}       % accent indigo for booktabs rules
\arrayrulecolor{tableRule}
\renewcommand{\arraystretch}{1.5}           % taller rows for breathing room
\setlength{\tabcolsep}{12pt}                % generous horizontal padding
\setlength{\heavyrulewidth}{1pt}            % top/bottom rule weight
\setlength{\lightrulewidth}{0.5pt}          % midrule weight

%% Tables sit left-aligned at the text margin with a touch of vertical air
%% above and below. We intentionally do NOT force full-width via
%% \extracolsep{\fill} — with only two columns it pushes the second column to
%% the page edge, looking unbalanced. Letting tables size to their content
%% renders better for typical markdown tables.
\AtBeginEnvironment{longtable}{%
  \setlength{\LTleft}{0pt}%
  \setlength{\LTright}{\fill}%
  \setlength{\LTpre}{8pt}%
  \setlength{\LTpost}{8pt}%
  \arrayrulecolor{tableRule}%
}
